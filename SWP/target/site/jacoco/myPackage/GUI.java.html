    public void showLoginScreen() {
        loop: while(true) {

            System.out.println("enter username:-");
            System.out.print(">>");
            String username = scanner.nextLine();
            System.out.println("enter password:-");
            System.out.print(">>");
            String password = scanner.nextLine();

            String msg = app.attemptLogin(username, password);
            switch (msg) {
                case "login successfully" -> {
                    app.Login(username,"user");
                    break loop;
                }
                case "logged in as admin" ->{
                    app.Login(username,"admin");
                    break loop;
                }
                case "wrong password" -> System.out.println("Incorrect password, try again ");
                case "wrong username" -> System.out.println("We cant find this user, try again");
                default -> throw new IllegalArgumentException("Unexpected value: " + msg);
            }
        }
    }

    private void addAppointment() {

        	while(true) {

            showWorkers();
            System.out.println("Choose worker");
            System.out.print(">> ");
            int workerIndex = Integer.parseInt(scanner.nextLine());
            Admin Admin = app.db.admins.get(workerIndex-1);

            System.out.println("Enter month (like: yy-mm-dd)");
            System.out.print(">> ");
            String date =scanner.nextLine();

            if(!checkValidDate(date)){
                System.out.println("invalid date");
                continue;
            }

                LocalDate Date = createDate(date);

                System.out.println("Enter time (like: '10 am' or '12 pm' from 7 am to 5 pm)");
            System.out.print(">> ");
            String time = scanner.nextLine();

            if(!checkValidTime(time)){
                System.out.println("invalid time");
                continue;
            }


                LocalTime Time = createTime(time);

                Appointment appointment=new Appointment();
            appointment.setId(5);
            appointment.setDate(Date);
            appointment.setTime(Time);
            appointment.setUser(app.LoggedUser);
            appointment.setAdmin(Admin);
            app.db.appointments.add(appointment);
            break;

        }

    public LocalTime createTime(String time) {
        String[] splitTime = time.trim().split(" ");
        int hour =Integer.parseInt(splitTime[0]);
        int minute=0;
        return LocalTime.of(hour,minute);
    }

    public LocalDate createDate(String date) {
        String[] splitDate = date.trim().split("-");
        int day =  Integer.parseInt(splitDate[2]);
        int month = Integer.parseInt(splitDate[1]);
        int year = Integer.parseInt(splitDate[0]);
        return LocalDate.of(year,month,day);
    }

    private void showWorkers() {
        int number=1;
        for(Admin w: app.db.admins){
            System.out.println(number+"- "+w.getUsername());
            number++;
        }
    }
    public boolean checkValidDate(String date) {
        String[] Date = date.trim().split("-");
        int day =  Integer.parseInt(Date[2]);
        int month = Integer.parseInt(Date[1]);
        int year = Integer.parseInt(Date[0]);

        if(day < 1 || day > 31 )
            return false;
        else if( month < 1 || month > 12)
            return false;
        else return year >= 2022;
    }

    public boolean checkValidTime(String time) {
        String[] Time = time.split(" ");
        int hour = Integer.parseInt(Time[0]);

        if(Time[1].equals("am"))
            return hour >= 7 && hour <= 11;
        else if(Time[1].equals("pm"))
            return hour == 12 || (hour >= 1 && hour <= 5);
        else
            return false;

    }
